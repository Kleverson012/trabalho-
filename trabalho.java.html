import java.util.ArrayList;
        import java.util.List;
        import java.util.Scanner;

class Question {
    String question;
    String[] options;
    int correctAnswerIndex;

    Question(String question, String[] options, int correctAnswerIndex) {
        this.question = question;
        this.options = options;
        this.correctAnswerIndex = correctAnswerIndex;
    }



    public static void main(String[] args) {
        List<Question> questions = new ArrayList<>();
        questions.add(new Question("Qual a capital da França?", new String[]{"A) Berlim", "B) Paris", "C) Madrid", "D) Lisboa"}, 1));
        questions.add(new Question("Qual é a maior montanha do mundo?", new String[]{"A) K2", "B) Everest", "C) Kangchenjunga", "D) Lhotse"}, 1));
        questions.add(new Question("Qual o elemento químico com símbolo 'O'?", new String[]{"A) Ouro", "B) Oxigênio", "C) Prata", "D) Hidrogênio"}, 1));
        questions.add(new Question("Qual a moeda do Japão?", new String[]{"A) Dólar", "B) Euro", "C) Iene", "D) Won"}, 2));
        questions.add(new Question("Quem escreveu 'Dom Casmurro'?", new String[]{"A) Machado de Assis", "B) Jorge Amado", "C) Clarice Lispector", "D) Guimarães Rosa"}, 0));
        questions.add(new Question("Qual o maior oceano do mundo?", new String[]{"A) Atlântico", "B) Índico", "C) Pacífico", "D) Ártico"}, 2));
        questions.add(new Question("Qual o planeta mais próximo do sol?", new String[]{"A) Vênus", "B) Terra", "C) Mercúrio", "D) Marte"}, 2));
        questions.add(new Question("Quem pintou a Mona Lisa?", new String[]{"A) Van Gogh", "B) Picasso", "C) Da Vinci", "D) Michelangelo"}, 2));
        questions.add(new Question("Qual a língua mais falada do mundo?", new String[]{"A) Inglês", "B) Mandarim", "C) Espanhol", "D) Árabe"}, 1));
        questions.add(new Question("Quem foi Albert Einstein?", new String[]{"A) Um matemático", "B) Um físico", "C) Um químico", "D) Um biólogo"}, 1));
        questions.add(new Question("Qual o continente onde está o Egito?", new String[]{"A) Europa", "B) Ásia", "C) África", "D) América"}, 2));
        questions.add(new Question("Qual o animal símbolo do Brasil?", new String[]{"A) Onça", "B) Lobo", "C) Águia", "D) Tatu"}, 0));
        questions.add(new Question("Em que ano o homem pisou na Lua?", new String[]{"A) 1965", "B) 1969", "C) 1972", "D) 1980"}, 1));
        questions.add(new Question("Qual o maior país do mundo?", new String[]{"A) Estados Unidos", "B) Canadá", "C) Rússia", "D) China"}, 2));
        questions.add(new Question("Qual a fórmula química da água?", new String[]{"A) H2O", "B) CO2", "C) O2", "D) NaCl"}, 0));

        int score = 0;
        Scanner scanner = new Scanner(System.in);

        for (Question q : questions) {
            System.out.println(q.question);
            for (String option : q.options) {
                System.out.println(option);
            }

            System.out.print("Digite a letra da sua resposta: ");
            String userAnswer = scanner.nextLine().toUpperCase();

            if (userAnswer.equals(q.options[q.correctAnswerIndex].substring(0, 1))) {
                System.out.println("Correto!");
                score++;
            } else {
                System.out.println("Incorreto! A resposta correta é " + q.options[q.correctAnswerIndex]);
            }
            System.out.println();
        }

        System.out.println("Você acertou " + score + " de " + questions.size() + " perguntas.");
        scanner.close();
    }
}
